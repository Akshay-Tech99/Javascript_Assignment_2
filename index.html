JAVASCRIPT DATA TYPES


1.)
let x = 5;
let y = x;
x = 10;
document.write(x);
document.write(y);

"5".

2.)

let obj2=obj1;
obj1.name="Bob";
document.write(obj1.name)

"Bob".
